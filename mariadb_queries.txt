
 
CREATE TABLE JOBS
(jobID INTEGER not NULL AUTO_INCREMENT,
 ownerID INTEGER not NULL,
 description VARCHAR(255),
 industry INTEGER,
 yearsExperience INTEGER,
 yearsEmployed INTEGER,
 educationLevel INTEGER,
 degreeRelevance INTEGER,
 certRelavance INTEGER,
 PRIMARY KEY(jobID)
 FOREIGN KEY(ownerID) REFERENCES owners(userID)
 ON UPDATE CASCADE ON DELETE RESTRICT)ENGINE=INNODB;
 
 CREATE TABLE SEEKERS 
(userID INTEGER not NULL AUTO_INCREMENT, 
 firstName VARCHAR(255)not NULL,
 lastName VARCHAR(255)not NULL,
 userName VARCHAR(255)not NULL, 
 password VARCHAR(255)not NULL,
 email VARCHAR(255)not NULL,
 contact VARCHAR(255),
 industry INTEGER not NULL,
 yearsExperience INTEGER,
 yearsEmployed INTEGER,
 educationLevel INTEGER,
 degreeRelevance INTEGER,
 certRelavance INTEGER,
 postDate DATE,
 PRIMARY KEY(userID)
 )ENGINE=INNODB;
 
 CREATE TABLE OWNERS 
(userID INTEGER not NULL AUTO_INCREMENT, 
 firstName VARCHAR(255)not NULL,
 lastName VARCHAR(255)not NULL,
 userName VARCHAR(255)not NULL, 
 password VARCHAR(255)not NULL,
 email VARCHAR(255)not NULL,
 contact VARCHAR(255)not NULL,
 company VARCHAR(255) not NULL,
 postDate DATE,
 PRIMARY KEY(userID)
 )ENGINE=INNODB;
 
 CREATE TABLE MATCHES
 ( matchID INTEGER not NULL AUTO_INCREMENT,
   userID INTEGER not NULL,
   jobID INTEGER not NULL,
   applied INTEGER not NULL,
   PRIMARY KEY(matchID),
   INDEX(userID),
   INDEX(jobID),
   FOREIGN KEY(userID) REFERENCES seekers(userID)
   ON UPDATE CASCADE ON DELETE RESTRICT,
   FOREIGN KEY(jobID) REFERENCES jobs(jobID)
   ON UPDATE CASCADE ON DELETE RESTRICT
 )ENGINE=INNODB;
 
 CREATE TABLE OWNED
 ( ownedID INTEGER not NULL AUTO_INCREMENT,
   userID INTEGER not NULL,
   jobID INTEGER not NULL,
   PRIMARY KEY(ownedID),
   INDEX(userID),
   INDEX(jobID),
   FOREIGN KEY(userID) REFERENCES owners(userID)
   ON UPDATE CASCADE ON DELETE RESTRICT,
   FOREIGN KEY(jobID) REFERENCES jobs(jobID)
   ON UPDATE CASCADE ON DELETE RESTRICT
 )ENGINE=INNODB;
